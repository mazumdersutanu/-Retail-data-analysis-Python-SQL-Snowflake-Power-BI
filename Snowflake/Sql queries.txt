--query 1 : Update the store table so that all stores have opening date on or after 1-Jan-2014, populate random date_from_parts

SELECT STOREID, STORENAME, STOREOPENINGDATE 
FROM DIMSTORE 
LIMIT 10;
SELECT DATEDIFF(DAY, '2014-01-01', CURRENT_DATE) AS TotalDays;
SELECT DATEADD(DAY, UNIFORM(0, 3000, RANDOM()), '2014-01-01') AS RandomDate;

UPDATE DIMSTORE SET STOREOPENINGDATE = DATEADD(DAY, UNIFORM(0, 3000, RANDOM()), '2014-01-01');
COMMIT;

--Query 4: We may have some orders in the Fact Table that may have a DateID which contains a value even before the store was opened.
--First we identify the records that have a problem.
--Identify a valid date that we can enter.
-- We perform the update.

DESC TABLE DIMDATE;

SELECT COUNT(*) AS BadOrders
FROM FACTORDERS F
JOIN DIMDATE D ON F.DATEID = D.DATEID
JOIN DIMSTORE S ON F.STOREID = S.STOREID
WHERE D.FULLDATE < S.STOREOPENINGDATE;

SELECT F.ORDERID, F.STOREID, S.STORENAME, S.STOREOPENINGDATE, D.FULLDATE AS ORDERDATE, 
DATEDIFF(DAY, S.STOREOPENINGDATE, CURRENT_DATE) * UNIFORM(1, 10, RANDOM()) * 0.1 AS CALCULATED_VALUE
FROM FACTORDERS F
JOIN DIMDATE D  ON F.DATEID  = D.DATEID
JOIN DIMSTORE S ON F.STOREID = S.STOREID
WHERE D.FULLDATE < S.STOREOPENINGDATE
ORDER BY F.ORDERID;
COMMIT;


--QUERY 5: List Customers who haven't placed an order in the last 30 days.

SELECT DISTINCT C.CUSTOMERID FROM DIMCUSTOMER C
JOIN FACTORDERS F  ON C.CUSTOMERID = F.CUSTOMERID
JOIN DIMDATE D ON F.DATEID = D.DATEID

--Query 6: List the store that was opened most recently along with its sales since then.

WITH store_rank AS 
(  
SELECT STOREID, STORENAME, STOREOPENINGDATE, ROW_NUMBER() OVER (ORDER BY STOREOPENINGDATE DESC) AS FINAL_RANK FROM DIMSTORE
),
most_recent_store AS 
(
SELECT STOREID FROM store_rank WHERE FINAL_RANK = 1
),
store_amount AS 
(
 SELECT O.STOREID, SUM(O.TOTALAMOUNT) AS TOTALAMOUNT  FROM FACTORDERS O JOIN most_recent_store S ON O.STOREID = S.STOREID
 GROUP BY O.STOREID
)
SELECT S.STOREID, S.STORENAME, S.STOREOPENINGDATE, S.REGION, A.TOTALAMOUNT
FROM DIMSTORE S
JOIN store_amount A ON S.STOREID = A.STOREID
ORDER BY A.TOTALAMOUNT DESC;

-- Query 7: Find customers who have ordered product from more than 3 categories in the last 6 months.

COMMIT;


--Query 8: Get the monthly total sales for the curent year.

SELECT D.MONTHNUMBER, D.MONTH AS MONTHNAME, SUM(O.TOTALAMOUNT) AS MONTHLY_AMOUNT FROM FACTORDERS O JOIN DIMDATE D ON O.DATEID = D.DATEID
WHERE D.YEARNUM = EXTRACT(YEAR FROM CURRENT_DATE)  
GROUP BY D.MONTHNUMBER, D.MONTH            
ORDER BY D.MONTHNUMBER; 

COMMIT;

--Query 9: Find the highest discount given on any order in the last 1 year.

WITH base_date AS 
(
SELECT O.ORDERID, O.CUSTOMERID, O.STOREID, O.ORDERAMOUNT, O.DISCOUNTAMOUNT,O.TOTALAMOUNT, D.FULLDATE  AS ORDERDATE,  ROW_NUMBER() OVER ( ORDER BY O.DISCOUNTAMOUNT DESC)     AS  DISCOUNTAMOUNT_RANK  FROM FACTORDERS O JOIN DIMDATE D ON O.DATEID = D.DATEID
WHERE D.FULLDATE >= DATEADD(YEAR, -1, CURRENT_DATE)
)
SELECT * FROM base_date WHERE DISCOUNTAMOUNT_RANK = 1;

--Query 11: Show the customer id of the Customer who has taken the maximum discount in their lifetime.
--Max Discount by Customer id in their lifetime



SELECT F.CUSTOMERID, C.FIRSTNAME, C.LASTNAME, SUM(F.DISCOUNTAMOUNT) AS TOTAL_DISCOUNT  FROM FACTORDERS F

--Query 12: List the customer who was placed maximum number of orders till date.
--List the customer who was placed maximum number of orders till date

WITH base_data AS 
(
SELECT CUSTOMERID, COUNT(ORDERID) AS ORDER_COUNT  FROM FACTORDERS
GROUP BY CUSTOMERID
), 
order_rank_data AS 
(
SELECT B.*, ROW_NUMBER() OVER ( ORDER BY ORDER_COUNT DESC ) AS ORDER_RANK FROM base_data B
)
SELECT O.CUSTOMERID, C.FIRSTNAME, C.LASTNAME, O.ORDER_COUNT
FROM order_rank_data O
JOIN DIMCUSTOMER C ON O.CUSTOMERID = C.CUSTOMERID
WHERE ORDER_RANK = 1;
COMMIT;

--Query 14: If the discount amount and the shipping cost was made static at 5 and 8% respectively will the sum of new total amount be greater than the total amount we have.
--If the discount amount and the shipping cost was made static at 5 and 8% respectively

--Query 16: Show the region category wise total amount for the last 6 months.

SELECT S.REGION, P.CATEGORY, SUM(F.TOTALAMOUNT) AS TOTAL_SALES,       
COUNT(F.ORDERID) AS TOTAL_ORDERS, ROUND(AVG(F.TOTALAMOUNT), 2) AS AVG_ORDER      
FROM FACTORDERS F
JOIN DIMDATE D ON F.DATEID = D.DATEID
JOIN DIMPRODUCT P ON F.PRODUCTID = P.PRODUCTID
JOIN DIMSTORE S ON F.STOREID  = S.STOREID
WHERE D.FULLDATE >= DATEADD(MONTH, -6, CURRENT_DATE)  
GROUP BY S.REGION, P.CATEGORY                         
ORDER BY S.REGION, TOTAL_SALES DESC;
COMMIT;

--Query 17: Show the top 5 products based on quantity ordered in the last 3 years.
 --show the top 5 products based on quantity ordered in the last 3 years.

WITH quantity_data AS
(
SELECT F.PRODUCTID, SUM(F.QUANTITYORDERED)  AS TOTAL_QUANTITY FROM FACTORDERS F JOIN DIMDATE D ON F.DATEID = D.DATEID
WHERE D.FULLDATE >= DATEADD(YEAR, -3, CURRENT_DATE)  
GROUP BY F.PRODUCTID
),
quantity_rank_data AS 
(
SELECT Q.*, ROW_NUMBER() OVER ( ORDER BY TOTAL_QUANTITY DESC) AS QUANTITY_WISE_RANK FROM quantity_data Q
)
SELECT R.PRODUCTID, P.PRODUCTNAME, P.CATEGORY, R.TOTAL_QUANTITY, R.QUANTITY_WISE_RANK
FROM quantity_rank_data R
JOIN DIMPRODUCT P ON R.PRODUCTID = P.PRODUCTID   
WHERE QUANTITY_WISE_RANK <= 5                    
ORDER BY QUANTITY_WISE_RANK;
COMMIT;

--Query 18: List the total amount for each loyalty program tier since year 2023.

SELECT P.PROGRAMNAME, P.PROGRAMTIER, SUM(F.TOTALAMOUNT) AS TOTAL_SALES, COUNT(F.ORDERID) AS TOTAL_ORDERS, ROUND(AVG(F.TOTALAMOUNT),2) AS AVG_SALE
FROM FACTORDERS F
JOIN DIMDATE D ON F.DATEID = D.DATEID
JOIN DIMCUSTOMER C ON F.CUSTOMERID = C.CUSTOMERID
JOIN DIMLOYALTYPROGRAM P ON C.LOYALTYPROGRAMID = P.LOYALTYPROGRAMID
WHERE D.YEARNUM >= 2023                     
GROUP BY P.PROGRAMNAME, P.PROGRAMTIER       
ORDER BY TOTAL_SALES DESC;
COMMIT;
--Query 19: Calculate the revenue generated by each store manager in june 2024

SELECT S.MANAGERNAME, S.STORENAME, S.REGION, SUM(F.TOTALAMOUNT)   AS TOTAL_SALES, COUNT(F.ORDERID) AS TOTAL_ORDERS, ROUND(AVG(F.TOTALAMOUNT), 2) AS AVG_SALE
FROM FACTORDERS F
JOIN DIMDATE D ON F.DATEID = D.DATEID
JOIN DIMSTORE S ON F.STOREID = S.STOREID
WHERE D.YEARNUM = 2024                 
AND D.MONTHNUMBER  = 6                    
GROUP BY S.MANAGERNAME, S.STORENAME, S.REGION
ORDER BY TOTAL_SALES DESC;
COMMIT;
-- Query 20: List the average order amount per store. along with the store name and type for the year 2024.

SELECT S.STORENAME, S.STORETYPE, S.REGION, ROUND(AVG(F.TOTALAMOUNT), 2) AS AVG_SALE, SUM(F.TOTALAMOUNT) AS TOTAL_SALES, COUNT(F.ORDERID) AS TOTAL_ORDERS      
FROM FACTORDERS F
JOIN DIMDATE D  ON F.DATEID  = D.DATEID
JOIN DIMSTORE S ON F.STOREID = S.STOREID
WHERE D.YEARNUM = 2024                        
GROUP BY S.STORENAME, S.STORETYPE, S.REGION   
ORDER BY AVG_SALE DESC;
COMMIT;
